@bot.message_handler(commands=['start'])
def handle_start(message):
    keyboard = telebot.types.ReplyKeyboardMarkup(True, True, True)
    keyboard.row('/start','/help')
    bot.send_message(message.from_user.id, "Добро пожаловать," + message.from_user.first_name, "Наберите нужное слово...", reply_markup=keyboard)
@bot.message_handler(commands=['help'])
def handle_start(message):
    hide = telebot.types.ReplyKeyboardHide()
    bot.send_message(message.from_user.id, "Наберите ваш термин и ждите ответа", reply_markup=hide)
